class Solution:
    def combinationSum(self, candidates: List[int], target: int) -> List[List[int]]:
        ans = []
        for i in range(len(candidates)):
            cur = candidates.pop()
            if target == cur:
                ans.append([cur])
            elif cur < target:
                subCombine = self.combinationSum(candidates + [cur], target - cur)
                for sub in subCombine:
                    sub.append(cur)
                ans += subCombine
        return ans
